package playersClasses;

import java.util.*;

public class PaulCrate extends Player {
	//declare variable
	private int pCWins;

	//constructors
	public PaulCrate() {
		name = "name not set";
	}

	public PaulCrate(String name) {
		this.name = name;
	}

	//setters
	public void setWins(int pCWins) {
		this.pCWins += pCWins;
	}

	public void setWins() {
		pCWins = 0;
	}

	@Override
	public void setName(String name) {
		this.name = name;

	}
	
	//getters
	public int getWins() {
		return pCWins;
	}


	@Override
	public String getName() {

		return name;
	}
	
	//abstract method will return a random answer
	@Override
	public String generateRoshambo() {
		int nextPlay = 0;
		Random rand = new Random();
		nextPlay = rand.nextInt(3);
		if (nextPlay == 0) {
			return playersClasses.Roshambo.rock.toString();
		} else if (nextPlay == 1) {
			return playersClasses.Roshambo.paper.toString();
		} else {
			return playersClasses.Roshambo.scissors.toString();
		}

	}


	
}
