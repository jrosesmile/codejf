package playersClasses;

public class Rocky extends Player {
	//declared variable
	private int rWins;

	//constructors
	public Rocky() {
		name = "name not set";
	}
	
	public Rocky(String name) {
		this.name = name;
	}

	//setters
	public void setWins(int rWins) {
		this.rWins += rWins;
	}

	public void setWins() {
		rWins = 0;
	}

	@Override
	public void setName(String name) {
		this.name = name;

	}
	
	//getters
	public int getWins() {
		return rWins;
	}

	@Override
	public String getName() {

		return name;
	}
	
	//abstract method always returns rock
	@Override
	public String generateRoshambo() {

		return playersClasses.Roshambo.rock.toString();
	}

	


}
